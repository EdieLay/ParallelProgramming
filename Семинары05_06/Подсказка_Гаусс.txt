for (k = 1; k <= n; k++) // прямой ход
{
	for (j = k + 1; j <= n; j++)
	{
		d = a[j][k] / a[k][k]; // формула (1)
		for (i = k; i <= n; i++)
		{
			a[j][i] = a[j][i] - d * a[k][i]; // формула (2)
		}
		b[j] = b[j] - d * b[k]; // формула (3)

	}
}

for (k = n; k >= 1; k--) // обратный ход
{
	d = 0;
	for (j = k + 1; j <= n; j++)
	{
		s = a[k][j] * x[j]; // формула (4)
		d = d + s; // формула (4)
	}
	x[k] = (b[k] - d) / a[k][k]; // формула (4)
}
